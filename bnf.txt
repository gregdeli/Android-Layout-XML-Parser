<layout> ::= <LinearLayout> | <RelativeLayout>

<LinearLayout> ::= '<LinearLayout' [android:id='<id>'] [android:orientation='<orientation>'] 'android:layout_width="<width>"' 'android:layout_height="<height>"' '>' <element>* '</LinearLayout>'

<RelativeLayout> ::= '<RelativeLayout' [android:id='<id>'] 'android:layout_width="<width>"' 'android:layout_height="<height>"' '>' <element>* '</RelativeLayout>'

<element> ::= <TextView> | <ImageView> | <Button> | <RadioGroup> | <ProgressBar> | <LinearLayout> | <RelativeLayout> | '<!--' <comment> '-->'

<TextView> ::= '<TextView' [android:id='<id>'] [android:textColor='<color>'] 'android:text="<text>"' 'android:layout_width="<width>"' 'android:layout_height="<height>"' '/>'

<ImageView> ::= '<ImageView' [android:id='<id>'] [android:padding='<padding>'] 'android:src="<src>"' 'android:layout_width="<width>"' 'android:layout_height="<height>"' '/>'

<Button> ::= '<Button' [android:id='<id>'] [android:padding='<padding>'] 'android:text="<text>"' 'android:layout_width="<width>"' 'android:layout_height="<height>"' '/>'

<RadioGroup> ::= '<RadioGroup' [android:id='<id>'] [android:checkedButton='<checkedButton>'] 'android:layout_width="<width>"' 'android:layout_height="<height>"' '>' <RadioButton>+ '</RadioGroup>'

<RadioButton> ::= '<RadioButton' [android:id='<id>'] 'android:text="<text>"' '/>'

<ProgressBar> ::= '<ProgressBar' [android:id='<id>'] [android:max='<max>'] [android:progress='<progress>'] 'android:layout_width="<width>"' 'android:layout_height="<height>"' '/>'

<comment> ::= <string>

<id> ::= <alphanumeric>

<text> ::= <alphanumeric>

<color> ::= <alphanumeric>

<padding> ::= <integer>

<width> ::= <alphanumeric> | <integer>

<height> ::= <alphanumeric> | <integer>

<orientation> ::= <alphanumeric>

<string> ::= '"' (<alphanumeric> | <specialChar>)* '"'

<integer> ::= <positiveInt> | <negativeInt>

<alphanumeric> ::= <letterOrDigit>+

<positiveInt> ::= <digit>+

<negativeInt> ::= '-' <positiveInt>

<letterOrDigit> ::= <letter> | <digit>

<letter> ::= 'a' | 'b' | ... | 'z' | 'A' | 'B' | ... | 'Z'

<digit> ::= '0' | '1' | ... | '9'

<specialChar> ::= any special character such as '.', ',', ':', ';', etc.
