<root> ::= "<LinearLayout" <linear-attrs> ">" <content> "</LinearLayout>"
          | "<RelativeLayout" <relative-attrs> ">" <content> "</RelativeLayout>"
          
<linear-attrs> ::= " android:layout_width=\"" <layout-width> "\""
                    " android:layout_height=\"" <layout-height> "\""
                    <optional-linear-attrs>
                    
<optional-linear-attrs> ::= " android:id=\"" <string> "\""
                            " android:orientation=\"" <string> "\""
                            | ε
                            
<relative-attrs> ::= <optional-string-attr>
                      " android:layout_width=\"" <layout-width> "\""
                      " android:layout_height=\"" <layout-height> "\""
                      
<optional-string-attr> ::= " android:id=\"" <string> "\""
                            | ε

<layout-width> ::= <string> | <integer>

<layout-height> ::= <string> | <integer>

<content> ::= <element> | <element> <content>

<element> ::= "<TextView" <text-view-attrs> "/>"
               | "<ImageView" <image-view-attrs> "/>"
               | "<Button" <button-attrs> "/>"
               | "<LinearLayout" <linear-attrs> ">" <content> "</LinearLayout>"
               | "<RelativeLayout" <relative-attrs> ">" <content> "</RelativeLayout>"

<text-view-attrs> ::= " android:text=\"" <string> "\""
                       <optional-string-attr>
                       " android:textColor=\"" <string> "\""
                       
<image-view-attrs> ::= " android:src=\"" <string> "\""
                       <optional-string-attr>
                       " android:padding=\"" <integer> "\""
                       
<button-attrs> ::= " android:text=\"" <string> "\""
                    <optional-string-attr>
